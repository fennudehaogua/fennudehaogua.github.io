链表总结

发表于 2018-09-03 | 分类于 JAVA

链表汇总

题目描述
1.链表的倒置（拆解法和头插法，头插法比较类似，就是创建了一个temp-head）

  • 备份next指针
  • 修改head.next
  • 移动head与new-head

2.链表的部分倒置
(从m到n，记录四个重要的节点，
pre-head倒置前头结点的前节点，
modify-list-tail倒置前的尾节点，
new-head相当于逆置后的head，
head相当于逆置后的尾节点的后继节点）

  • 将head向后移动m-1个位置到达逆置前的头结点，记录该节点以及该节点的前驱节点
  • 从head开始，逆置n-m+1个节点
  • 将pre-head与new-head连接，modify-list-tail与head连接

3.判断链表是否有环，环的起点，环的长度

  • 快慢指针
  • 当两个指针相遇就代表有环
  • 相遇的时候将快指针移动至head位置，两个指针再同时以相同的速度移动，再相遇的位置即为环的起点
  • 快慢指针相遇后，一个指针不懂，另一个移动直到两个指针相遇即为环的长度

4.链表的划分：已知链表头指针head与数值x，小于的放前面，大于的放后面，保持相对位置

  • 新创建两个头结点，遍历链表，一个存小的一个存大的，再将两个链表连接起来

5.复杂链表的深度拷贝

  • Hashmap存储Random节点
  • 在每个节点N的后面创建复制的节点N’，复制连接，然后拆分成两个链表。奇数位上的连起来就是原链表，偶数位的链接起来就是复制的链表。

6.排序链表的合并

  • 遍历两个链表，逐渐对比两个链表的数值取小的插入新链表中即可。

7.多个排序链表的合并

//链表的倒置
public ListNode ReverseList(ListNode head){
    ListNode new_head = null;

    while (head != null){
        ListNode next = head.next;
        head.next = new_head;
        new_head = head;
        head = next;
    }
    return new_head;
}
//头插法
public ListNode ReverseList_headinsert(ListNode head){
    ListNode temp_head = new ListNode(0);
    while (head != null){
        ListNode next = head.next;
        head.next = temp_head.next;
        temp_head.next = head;
        head = next;
    }
    return temp_head.next;

}

//链表的部分倒置
public ListNode ReverseList1(ListNode head,int m,int n){
    int change_len = n - m + 1;
    ListNode pre_head = null;
    ListNode result = head;
    while ((head != null) && (--m > 0)){
        pre_head = head;
        head = head.next;
    }



    ListNode modify_list_tail = head;
    ListNode new_head = null;
    while ((head != null) && (change_len>0)){
        ListNode next = head.next;
        head.next = new_head;
        new_head = head;
        head = next;
        change_len--;
    }

    modify_list_tail.next = head;

    if (pre_head==null){
        return new_head;
    }else {
        pre_head.next = new_head;

    }

return result;
}

//求链表是否有环
public boolean jiaoDian(ListNode head){
    ListNode slow = head;
    ListNode fast = head;
    if (fast.next == null){
        return false;
    }
    while (fast.next != null){
        slow = slow.next;
        fast = fast.next.next;
        if (fast == null){
            return false;
        }
        if (slow.val == fast.val){
            break;
        }

    }
    return true;
}

//求链表有环的话环的起点
public ListNode start_Node(ListNode head){
    ListNode slow = head;
    ListNode fast = head;
    if (fast.next == null){
        return null;
    }
    while (fast.next != null){
        slow = slow.next;
        fast = fast.next.next;
        if (fast == null){
            return null;
        }
        if (slow.val == fast.val){
            break;
        }
    }
    fast = head;
    while (fast != slow){
        fast = fast.next;
        slow = slow.next;
    }
    return fast;
}
//求链表有环的话环的长度
public int length_circle(ListNode head){
    ListNode slow = head;
    ListNode fast = head;
    if (fast.next == null){
        return 0;
    }
    while (fast.next != null){
        slow = slow.next;
        fast = fast.next.next;
        if (fast == null){
            return 0;
        }
        if (slow.val == fast.val){
            break;
        }
    }
    slow = slow.next;
    int length = 1;
    while (fast != slow){
        slow = slow.next;
        length++;
    }
    return length;
}


//划分链表
public ListNode partition_Node(ListNode head,int target){
    ListNode less_head = new ListNode(0);
    ListNode more_head = new ListNode(0);
    ListNode less_head1 = less_head;
    ListNode more_head1 = more_head;
    while (head != null){
        if (head.val < target){
            less_head.next = head;
            less_head = less_head.next;
        }else {
            more_head.next = head;
            more_head = more_head.next;
        }
        head = head.next;
    }
    more_head.next = null;
    less_head.next = more_head1.next;
    return less_head1.next;
}



//链表的深度拷贝
//两种方法 Hashmap 和 每个节点N后面加上N'，然后定位链接，最后拆分为两个链表，奇数位置为原始链表，偶数位置为复制位置。

public RandomListNode clone(RandomListNode pHead){

    if(pHead == null){
        return null;
    }
    HashMap<RandomListNode, RandomListNode> map = new HashMap<>();
    RandomListNode pClonedHead = new RandomListNode(pHead.val); //复制链表的头结点
    RandomListNode pNode = pHead, pClonedNode = pClonedHead;
    map.put(pNode, pClonedNode);

    //第一步，hashMap保存，原链表节点映射复制链表节点
    while(pNode.next != null){
        pClonedNode.next = new RandomListNode(pNode.next.val);
        pNode = pNode.next;
        pClonedNode = pClonedNode.next;
        map.put(pNode, pClonedNode);
    }

    //第二步：找到对应的randomListNode
    pNode = pHead;
    pClonedNode = pClonedHead;
    while(pClonedNode!=null){
        pClonedNode.randomListNode = map.get(pNode.randomListNode);
        pNode = pNode.next;
        pClonedNode = pClonedNode.next;
    }
    return pClonedHead;
}


//两个有序链表的合并
public ListNode combine_two_sorted_listnode(ListNode listNode1,ListNode listNode2){
    ListNode result = new ListNode(0);
    ListNode result1 = result;
    while (listNode1 != null && listNode2 != null){
        if (listNode1.val < listNode2.val){
            result1.next = listNode1;
            listNode1 = listNode1.next;
        }else {
            result1.next = listNode2;
            listNode2 = listNode2.next;
        }
        result1 = result1.next;
    }

    if (listNode1 == null){
        result1.next = listNode2;
    }
    if (listNode2 == null){
        result1.next = listNode1;
    }
    return result.next;

}

//多个有序链表的合并
//方法一：优先队列
//基于优先级队列，我们可以将K个链表的头结点全部添加到队列，
// 由于优先级队列采用了最小堆数据结构，堆顶为队列的最小元素，
// 我们将其取出添加到结果链表中，取出元素对应的链表下移一个节点，并将这个节点添加到优先级队列中；
// 然后我们继续取出堆顶元素，...，直到优先级队列为空，那么其中所有元素取尽，
// K个链表的元素已经全部排序到结果链表。

//方法二
//归并排序”（Merging Sort）：将两个或两个以上的有序表组合成一个新的有序表，
// 无论是顺序存储结构还是链式存储结构，对于任何两个长度分别为m和n的有序表，
// 其组合都可在O(m+n)的时间复杂度量级上完成。对于K个有序表，假设共有N个元素，
// 且这些有序表初始状态都不为空，每个有序表平均拥有N/K个元素。
// 最常用的方法是采用“二分”的思想进行两两合并：
// 第一轮循环，有序表lists[0]与lists[(K+1)/2]，lists[1]与lists[(K+1)/2+1]，
// lists[2]与lists[(K+1)/2+2]....，lists[K/2-1]与lists[K-1]。
// 这样K个有序表就被组合成了K/2个有序表；第二轮循环后将减少为K/4个有序表；
// 直到组合成一个具有N个元素的有序表。总的时间复杂度：O(NKlogK)。
//

//详见：https://blog.csdn.net/jin_kwok/article/details/51582176
public ListNode mergeKLists(ListNode[] lists ){
    int len=lists.length;

    if(lists==null||len==0)
        return null;
    if(len==1)
        return lists[0];

    while(len>1)//基于“二分”思想进行链表的两两组合
    {
        int mid=(len+1)/2;//二分
        for(int i=0;i<len/2;i++)
        {
            lists[i]=mergeTwoLists(lists[i],lists[i+mid]);
        }
        len=mid;
    }
    return lists[0];
}

public ListNode mergeTwoLists(ListNode L1,ListNode L2)
{
    if(L1==null)return L2;
    if(L2==null)return L1;

    ListNode head=new ListNode(0);//保存结果的链表，头结点初始化
    ListNode phead=head;

    while(L1!=null&&L2!=null)//两个链表归并排序
    {
        if(L1.val <=L2.val)
        {
            phead.next=L1;//接入结果链表
            phead=phead.next;//移动指针
            L1=L1.next;
        }
        else
        {
            phead.next=L2;
            phead=phead.next;
            L2=L2.next;
        }
    }
    if(L1!=null)
        phead.next=L1;
    else
        phead.next=L2;

    return head.next;//初始化的第一个节点不属于结果
}




public ListNode mergeKLists1(ListNode[] lists)
{
    if(lists==null||lists.length==0)
        return null;
    if(lists.length==1)
        return lists[0];


    PriorityQueue<ListNode> PQ=new PriorityQueue<>();

    ListNode head=new ListNode(0);
    ListNode phead=head;

    for(ListNode list:lists)
    {
        if(list!=null)
            PQ.offer(list);
    }
    while(!PQ.isEmpty())
    {
        ListNode temp=PQ.poll();
        phead.next=temp;
        phead=phead.next;

        if(temp.next!=null)
            PQ.add(temp.next);
    }
    return head.next;
}
